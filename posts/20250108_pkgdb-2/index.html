match_emails_in_authorsR <- function(str) {
  
  pattern_email    <- "[\"|'|[ ]|[<]]([^\"]+?)[@]{1}(.+?)[\"|'|[ ]|[>]]"
  resultat <- stringi::stri_match_all_regex(
              str = str, 
              pattern = pattern_email,
              omit_no_match = FALSE)[[1]]
  
  return(resultat)

}

match_tld_in_email_domain <- function(str) {
  
  pattern_tld    <- "([^[.]]+$)"
  resultat <- stringi::stri_match_all_regex(
              str = str, 
              pattern = pattern_tld,
              omit_no_match = FALSE)[[1]]
  
  return(resultat[,2])

}
