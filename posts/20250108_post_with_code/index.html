  1. fonction extract_email
extract_email <- function(text) {

  regex_email    <- "[\"]([^[ ]|[\"]]+?[@]{1}.+?)[\"]"
  
  resultat <- stringi::stri_extract_all(text, regex = regex_email) |>
    unlist() |> 
    na.omit() |> 
    gsub(pattern = "[\"]", replacement = "") |> 
    as_tibble() |> 
    rename(email = value)
  
  return(resultat)
  
}
